import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;
import modelo.*;
import servico.BibliotecaService;

public class BibliotecaServiceTest {
    
    private BibliotecaService biblioteca;
    private Livro livroTeste;
    private Usuario usuarioTeste;
    
    @BeforeEach
    public void setup() {
        biblioteca = new BibliotecaService();
        livroTeste = new Livro("Teste", "Autor", "T001", "123456", 100, "Editora");
        usuarioTeste = new Usuario("Teste User", "teste@test.com", "000.000.000-00");
    }
    
    @Test
    public void testarAdicionarMaterial() {
        biblioteca.adicionarMaterial(livroTeste);
        assertEquals(1, biblioteca.getTotalMateriais());
    }
    
    @Test
    public void testarRegistrarUsuario() {
        biblioteca.registrarUsuario(usuarioTeste);
        assertEquals(1, biblioteca.getTotalUsuarios());
    }
    
    @Test
    public void testarBuscarMaterialPorId() {
        biblioteca.adicionarMaterial(livroTeste);
        var encontrado = biblioteca.buscarMaterialPorId("T001");
        assertTrue(encontrado.isPresent());
        assertEquals("Teste", encontrado.get().getTitulo());
    }
    
    @Test
    public void testarBuscarMaterialPorTermo() {
        biblioteca.adicionarMaterial(livroTeste);
        var resultados = biblioteca.buscarMaterialPorTermo("Teste");
        assertEquals(1, resultados.size());
    }
    
    @Test
    public void testarRealizarEmprestimo() {
        biblioteca.adicionarMaterial(livroTeste);
        biblioteca.registrarUsuario(usuarioTeste);
        
        boolean sucesso = biblioteca.realizarEmprestimo("T001", "Teste User");
        assertTrue(sucesso);
        assertTrue(livroTeste.isEmprestado());
    }
    
    @Test
    public void testarRealizarDevolucao() {
        biblioteca.adicionarMaterial(livroTeste);
        biblioteca.registrarUsuario(usuarioTeste);
        biblioteca.realizarEmprestimo("T001", "Teste User");
        
        boolean sucesso = biblioteca.realizarDevolucao("T001", "Teste User");
        assertTrue(sucesso);
        assertFalse(livroTeste.isEmprestado());
    }
    
    @Test
    public void testarPodeSerEmprestado() {
        assertTrue(livroTeste.podeSerEmprestado());
        livroTeste.registrarEmprestimo("Usuario");
        assertFalse(livroTeste.podeSerEmprestado());
    }
    
    @Test
    public void testarBuscaMateriaDisponivel() {
        biblioteca.adicionarMaterial(livroTeste);
        biblioteca.registrarUsuario(usuarioTeste);
        
        assertEquals(1, biblioteca.listarMateriasDiponiveis().size());
        
        biblioteca.realizarEmprestimo("T001", "Teste User");
        assertEquals(0, biblioteca.listarMateriasDiponiveis().size());
    }
    
    @Test
    public void testarEmprestimoAtrasado() {
        Emprestimo emp = new Emprestimo("T001", "Teste", 1);
        assertFalse(emp.estaAtrasado()); // hoje nao esta atrasado
    }
}

TESTE DE INTEGRACAO

@Test
public void testarFluxoCompletoEmprestimo() {
    // Setup
    Livro livro1 = new Livro("Clean Code", "Martin", "L001", "123", 300, "Ed1");
    Livro livro2 = new Livro("Refactoring", "Fowler", "L002", "456", 250, "Ed2");
    Usuario user1 = new Usuario("Alice", "alice@test.com", "111.111.111-11");
    Usuario user2 = new Usuario("Bob", "bob@test.com", "222.222.222-22");
    
    biblioteca.adicionarMaterial(livro1);
    biblioteca.adicionarMaterial(livro2);
    biblioteca.registrarUsuario(user1);
    biblioteca.registrarUsuario(user2);
    
    // Validar estado inicial
    assertEquals(2, biblioteca.getTotalMateriais());
    assertEquals(2, biblioteca.getTotalUsuarios());
    assertEquals(2, biblioteca.listarMateriasDiponiveis().size());
    
    // Emprestar
    assertTrue(biblioteca.realizarEmprestimo("L001", "Alice"));
    assertTrue(biblioteca.realizarEmprestimo("L002", "Bob"));
    
    // Validar apos emprestimo
    assertEquals(1, biblioteca.listarMateriasDiponiveis().size());
    assertEquals(2, biblioteca.getTotalEmprestimosAtivos());
    
    // Devolver um
    assertTrue(biblioteca.realizarDevolucao("L001", "Alice"));
    
    // Validar apos devolucao
    assertEquals(2, biblioteca.listarMateriasDiponiveis().size());
    assertEquals(1, biblioteca.getTotalEmprestimosAtivos());
}
